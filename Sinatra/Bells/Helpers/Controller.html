def to_render_hash(hash = {}, compact = true)
  if hash.is_a?(Array)
    ary, hash, at = hash, {}, '@'

    ary.each { |key| hash[key] = instance_variable_get(
      (name = key.to_s).start_with?(at) ? name : "#{at}#{name}") }
  end

  compact ? hash.delete_if { |_, value| !value } : hash
end
