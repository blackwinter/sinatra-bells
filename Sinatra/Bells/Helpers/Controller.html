def self.included(base)
  error_code = Rack::Utils::HTTP_STATUS_CODES

  base.class_eval {
    [400, 401, 403, 404, 500].each { |code|
      error(code) {
        @code, @error = code, error_code[code]
        render_error
      }
    }

    [400, 401, 403].each { |code|
      name = error_code[code].downcase.tr(' ', '_')
      define_method(name) { |body = nil| error(code, body) }
    }
  }
end

def to_render_hash(hash = {}, compact = true)
  if hash.is_a?(Array)
    ary, hash, at = hash, {}, '@'

    ary.each { |key| hash[key] = instance_variable_get(
      (name = key.to_s).start_with?(at) ? name : "#{at}#{name}") }
  end

  compact ? hash.delete_if { |_, value| !value } : hash
end
